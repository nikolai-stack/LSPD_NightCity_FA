from bs4 import BeautifulSoup # Load HTML content soup = BeautifulSoup(html, 'html.parser') # Find the section title for "Hiérarchie" hierarchie_header = soup.find('div', class_='nav-section-title', string='Hiérarchie') if hierarchie_header: # Find the next sibling that contains the content (usually the section content div) content_section = hierarchie_header.find_parent('div', class_='nav-section').find_next_sibling('div', class_='cards-grid') if content_section: # Create a new container for a nicer layout (we'll use a flexbox with spacing) new_container = soup.new_tag('div', style="display: flex; flex-wrap: wrap; gap: 30px; justify-content: center; margin: 30px 0;") # Add a header logo at the top center header_logo = soup.new_tag('img', src="https://upload.wikimedia.org/wikipedia/commons/thumb/4/4e/Los_Santos_Police_Department_Logo.png/240px-Los_Santos_Police_Department_Logo.png", style="width:120px; height:120px; margin-bottom:20px;") new_container.append(header_logo) # Iterate over each existing card in the section and reformat them for card in content_section.find_all('div', class_='group-card'): # Wrap each card in a container card_container = soup.new_tag('div', style="background: rgba(201,168,76,0.1); border-radius: 12px; padding:20px; box-shadow: 0 4px 8px rgba(0,0,0,0.15); transition: transform 0.2s; width:280px; text-align:center;") card_container.append(card) # Add small LSPD logos inside each card small_logo = soup.new_tag('img', src="https://upload.wikimedia.org/wikipedia/commons/thumb/4/4e/Los_Santos_Police_Department_Logo.png/100px-Los_Santos_Police_Department_Logo.png", style="width:40px; height:40px; margin-bottom:10px;") card_container.insert(0, small_logo) # Add subtle hover effect (scale up) card['style'] = 'transition: transform 0.2s;' card_container['onmouseover'] = 'this.style.transform="scale(1.05)";' card_container['onmouseout'] = 'this.style.transform="scale(1)";' # Insert the card container into the new layout new_container.append(card_container) # Replace the original content section with the new layout content_section.replace_with(new_container) # Save modified HTML output_hierarchy_path = Path('/mnt/data/manuele_lspd_hierarchy_modifie.html') output_hierarchy_path.write_text(str(soup), encoding='utf-8') print(f"Hiérarchie améliorée enregistrée : {output_hierarchy_path}")
